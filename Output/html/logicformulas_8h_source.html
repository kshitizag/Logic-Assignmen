77char evaluate(struct node* root, struct atom atoms[]){
78 // return the value of an atom if the value of the node is an atom
79 if (isAtom(root->item)) return valueOfAtom(root->item, atoms);
80 char lval, rval;
81
82 // for an operator, evaluate the children subtrees of the node and perform the corresponding operation
83 if (root->item != '~'){
84 lval = evaluate(root->left, atoms);
85 }
86 rval = evaluate(root->right, atoms);
87 switch (root->item)
88 {
89 case '~':
90 return NOT(rval);
91 break;
92
93 case '*':
94 return AND(lval, rval);
95 break;
96
97 case '+':
98 return OR(lval, rval);
99 break;
100
101 case '>':
102 return IMPL(lval, rval);
103 break;
104 }
105}
